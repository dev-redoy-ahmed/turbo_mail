## One-Line Installation Command:
sudo apt update && sudo apt install -y redis-server && sudo systemctl enable redis-server && sudo systemctl start redis-server && redis-cli ping && echo "‚úÖ Redis installed successfully!"

## Step-by-Step Commands:

# 1. Update system
sudo apt update && sudo apt upgrade -y

# 2. Install Redis
sudo apt install redis-server -y

# 3. Enable auto-start
sudo systemctl enable redis-server

# 4. Start Redis service
sudo systemctl start redis-server

# 5. Test Redis
redis-cli ping

## Redis Status Check Commands:

# Check if Redis is running
sudo systemctl status redis-server

# Test Redis connection
redis-cli ping

# Check Redis version
redis-cli --version

# Monitor Redis activity
redis-cli monitor

# Check Redis process
ps aux | grep redis

# Check Redis port
sudo netstat -tlnp | grep 6379

# View Redis logs
sudo journalctl -u redis-server -f

## Redis Management Commands:

# Restart Redis
sudo systemctl restart redis-server

# Stop Redis
sudo systemctl stop redis-server

# Check Redis memory usage
redis-cli info memory

# Check Redis configuration
sudo nano /etc/redis/redis.conf

## Quick Health Check:
redis-cli ping && echo "‚úÖ Redis is working!" || echo "‚ùå Redis is not responding"
